export function useClock(updateInterval: number = 1000): ClockTime {
  const [time, setTime] = useState<ClockTime>(getCurrentTime());

  useEffect(() => {
    const timer = setInterval(() => {
      setTime(getCurrentTime());
    }, updateInterval);

    return () => clearInterval(timer);
  }, [updateInterval]);

  return time;
}
